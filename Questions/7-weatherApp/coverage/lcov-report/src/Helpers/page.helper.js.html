import { connect } from "react-redux";
import { get } from 'lodash';
 
 
function createPage(Component, key = '', actions = {}) {
    const mapStateToProps = state => ({ page: get(state, key) });
    return connect(mapStateToProps, actions)(Component);
}
 
export {
    createPage
};
