export default (Chart) => {
  
  function createTooltipAt(chart, index) {
    for (let sector of chart.getDatasetMeta(index).data) {
      chart.pluginTooltips.push(createNewTooltip(chart, sector));
    }
  }
  
  function createNewTooltip(chart, sector) {
    const settings = {
      _chart: chart.chart,
      _chartInstance: chart,
      _data: chart.data,
      _options: chart.options.tooltips,
      _active: [sector]
    };
  
    return new Chart.Tooltip(settings, chart);
  }
  
  function beforeRender(chart) {
    chart.pluginTooltips = [];
  
    chart.config.data.datasets.forEach((_, index) => { createTooltipAt(chart, index) });
  
    chart.options.tooltips.enabled = false;
  }
  
  function tooltipOverrideStyle(tooltip) {
    const fontColor = '#585858';
    const { _options: options} = tooltip;
  
    let newOptions = {
      ...options,
      backgroundColor: 'rgba(0,0,0,0)',
      bodyFontColor: fontColor,
      titleFontColor: fontColor,
      titleFontFamily: `'Montserrat', sans-serif`,
      displayColors: false,
      bodyFontSize: tooltip._chart.height * 0.06
    }
  
    return newOptions;
  }
  
  function afterDraw(chart, easing) {
    if (!chart.allTooltipsOnce) {
      if (easing !== 1) return;
      chart.allTooltipsOnce = true;
    }
  
    chart.options.tooltips.enabled = true;
    Chart.helpers.each(chart.pluginTooltips, (tooltip) => {
      tooltip.initialize();
      tooltip._options = tooltipOverrideStyle(tooltip);
      tooltip.update();
      tooltip.transition(easing).draw();
    });
    chart.options.tooltips.enabled = false;
  }
  
 
  Chart.plugins.register({ beforeRender, afterDraw });
}
